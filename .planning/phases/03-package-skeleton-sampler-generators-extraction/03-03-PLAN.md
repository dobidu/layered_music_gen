def generate_random_key(rng: random.Random) -> str:
    """Weighted random musical key. 1 rng.random() call, 24-branch threshold loop with explicit fallback."""
    key_ranges = [
        (0.107, 'G'), (0.209, 'C'), (0.296, 'D'), (0.357, 'A'), (0.417, 'C#'), (0.47, 'F'),
        (0.518, 'Am'), (0.561, 'G#'), (0.603, 'Em'), (0.645, 'Bm'), (0.681, 'E'), (0.716, 'A#'),

    for prob, key in key_ranges:

def generate_random_swing(rng: random.Random) -> float:
    """2 draws always: rng.choices(...) + rng.uniform(-0.02, 0.02); clamped to [0.5, 0.75]."""
    swing_weights = [(0.5, 0.3), (0.66, 0.5), (0.75, 0.2)]
    base_swing = rng.choices(
        [s[0] for s in swing_weights],
        weights=[s[1] for s in swing_weights],
    )[0]
    variation = rng.uniform(-0.02, 0.02)
    return min(0.75, max(0.5, base_swing + variation))

class SongParams:
    """Frozen song-level parameters (D-20). Construct via `SongParams.sample(rng, cfg)`."""
    key: str
    tempo: int

    time_signature_variation: float

    @classmethod
    def sample(
        cls,
        rng: random.Random,
        cfg: Optional[config.Config] = None,
        *,
        time_signature_variation: float = 1.0,
    ) -> "SongParams":
        """Draw all song-level parameters in the canonical order (D-21).

        RNG draw order (must match music_gen.py:1010-1024 verbatim — RESEARCH Risk #2):
          1. generate_random_key(rng)
          2. generate_random_tempo(rng)
          3. generate_random_time_signature(rng)
          4. generate_random_swing(rng)
          5. generate_song_arrangement(rng, cfg.song_structures_file)
          6. LOOP UNTIL validate_measures_dict(...):
                 generate_song_measures(ts_base, ts_var, rng)

        key = generate_random_key(rng)
        tempo = generate_random_tempo(rng)
        time_sig_base = generate_random_time_signature(rng)

        unique_parts, arrangement = generate_song_arrangement(

        )
        while True:

                time_sig_base, time_signature_variation, rng,
            )
            if validate_measures_dict(measures, signatures):
                break
        return cls(
            key=key,
            tempo=tempo,
            time_signature_base=time_sig_base,
            time_signature_variation=time_signature_variation,
            swing_amount=swing,
            signatures_per_part=signatures,

class TestSongParamsSample:
    def test_sample_returns_songparams(self):
        p = SongParams.sample(random.Random(42))
        assert isinstance(p, SongParams)

        a = SongParams.sample(random.Random(42))

def _bare_random_calls(source: str) -> List[ast.Call]:
    """AST walk that returns `random.<attr>(...)` Call nodes, excluding `random.Random(...)` constructor."""
    tree = ast.parse(source)
    hits = []
    for node in ast.walk(tree):
        if (
            isinstance(node, ast.Call)
            and isinstance(node.func, ast.Attribute)
            and isinstance(node.func.value, ast.Name)
            and node.func.value.id == "random"
            and node.func.attr != "Random"  # `random.Random(seed)` is OK — it's the RNG constructor, not a bare draw
        ):
            hits.append(node)
    return hits


def test_no_bare_random_in_sampler():
    with open(SAMPLER_PATH, "r") as f:
        source = f.read()
    hits = _bare_random_calls(source)
    assert hits == [], (
        f"Found {len(hits)} bare random.<method>() call(s) in sampler.py at line(s) "
